🔗 Conectar ao Google Sheets

Para o painel funcionar, você precisa criar uma planilha no Google Sheets e publicar um Apps Script como API. Siga os passos:

1
Acesse sheets.google.com e crie uma nova planilha chamada Decole - Dados
2
Crie uma aba chamada cursos e outra chamada modulos
3
No menu, vá em Extensões → Apps Script e cole o código abaixo
4
Clique em Implantar → Nova implantação → App da Web — acesso para Qualquer pessoa
5
Copie a URL gerada e cole no campo abaixo
// Cole este código no Google Apps Script // Após colar: Implantar > Gerenciar implantações > editar > Nova versão > Implantar function doGet(e) { const action = e.parameter.action; const callback = e.parameter.callback; // JSONP support const ss = SpreadsheetApp.getActiveSpreadsheet(); function out(obj) { const json = JSON.stringify(obj); const body = callback ? callback + '(' + json + ')' : json; const mime = callback ? ContentService.MimeType.JAVASCRIPT : ContentService.MimeType.JSON; return ContentService.createTextOutput(body).setMimeType(mime); } // ── LEITURA ────────────────────────────────────── if (action === 'getCursos') { const sheet = ss.getSheetByName('cursos'); if (!sheet || sheet.getLastRow() < 2) return out({ok:true, data:[]}); const [headers, ...rows] = sheet.getDataRange().getValues(); return out({ok:true, data: rows.map(r => { const o = {}; headers.forEach((h,i) => o[h] = r[i]); return o; })}); } if (action === 'getModulos') { const sheet = ss.getSheetByName('modulos'); if (!sheet || sheet.getLastRow() < 2) return out({ok:true, data:[]}); const [headers, ...rows] = sheet.getDataRange().getValues(); return out({ok:true, data: rows.map(r => { const o = {}; headers.forEach((h,i) => o[h] = r[i]); return o; })}); } // ── ESCRITA via GET + payload ───────────────────── const raw = e.parameter.payload; if (!raw) return out({ok:false, error:'missing payload'}); const data = JSON.parse(decodeURIComponent(raw)); if (action === 'saveCurso') { const sheet = ss.getSheetByName('cursos'); const HEADERS = ['id','nome','descricao','paraQuem','duracao','mensalidade', 'valorTotal','materialTotal','materialVista','materialParcelado', 'slug','updatedAt']; if (sheet.getLastRow() === 0) sheet.appendRow(HEADERS); const allData = sheet.getDataRange().getValues(); const ids = allData.map(r => String(r[0])); const rowIndex = ids.indexOf(String(data.id)); const row = HEADERS.map(h => data[h] !== undefined ? data[h] : ''); if (rowIndex > 0) { sheet.getRange(rowIndex + 1, 1, 1, row.length).setValues([row]); } else { sheet.appendRow(row); } return out({ok:true}); } if (action === 'saveModulos') { const sheet = ss.getSheetByName('modulos'); const HEADERS = ['id','cursoId','titulo','descricao','ordem']; if (sheet.getLastRow() === 0) sheet.appendRow(HEADERS); const allData = sheet.getDataRange().getValues(); for (let i = allData.length - 1; i >= 1; i--) { if (String(allData[i][1]) === String(data.cursoId)) sheet.deleteRow(i + 1); } data.modulos.forEach((mod, i) => { sheet.appendRow([mod.id, data.cursoId, mod.titulo, mod.descricao || '', i + 1]); }); return out({ok:true}); } if (action === 'deleteCurso') { const sheet = ss.getSheetByName('cursos'); const allData = sheet.getDataRange().getValues(); const ids = allData.map(r => String(r[0])); const rowIndex = ids.indexOf(String(data.id)); if (rowIndex > 0) sheet.deleteRow(rowIndex + 1); const modSheet = ss.getSheetByName('modulos'); if (modSheet && modSheet.getLastRow() > 1) { const modData = modSheet.getDataRange().getValues(); for (let i = modData.length - 1; i >= 1; i--) { if (String(modData[i][1]) === String(data.id)) modSheet.deleteRow(i + 1); } } return out({ok:true}); } return out({ok:false, error:'action not found'}); }
